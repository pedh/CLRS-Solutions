   #+BEGIN_SRC
   HEAP-INCREASE-KEY(A, i, key)
       if key < A[i]
           error "new key is smaller than current key"
       A[i] = key
       while i > 1 and A[PARENT(i)] < A[i]
           A[PARENT(i)] = A[i]
           i = PARENT(i)
       A[i] = key
   #+END_SRC
